 Работа с вкладками                 ; 
Ctrl + N	                        ; Создать новый файл
Ctrl + O	                        ; Открыть файл...
Ctrl + S	                        ; Сохранить текущий файл
Ctrl + Shift + S	                ; Сохранить как...
Ctrl + W	                        ; Закрыть текущую вкладку
Ctrl + Shift + T                	; Восстановить последнюю закрытую вкладку
Ctrl + Tab	                        ; Переключиться на следующую вкладку
Ctrl + Shift + Tab              	; Переключиться на предыдущую вкладку
Ctrl + [1-9]                    	; Перейти к вкладке 1–9
Ctrl + K Ctrl + O               	; Открыть папку проекта

                                    ;
Навигация и поиск                   ; 
Ctrl + P	                        ; Быстрый поиск и открытие файлов
Ctrl + G	                        ; Перейти к строке (введите номер)
Ctrl + F                        	; Найти в файле
Ctrl + H	                        ; Найти и заменить в файле
Ctrl + Shift + F	                ; Найти во всех файлах проекта
Ctrl + Shift + H	                ; Заменить во всех файлах проекта
Alt + F12	                        ; Просмотреть определение (Peek Definition)
Alt + ⬅️ / ➡️	                    ; Навигация назад / вперёд по истории

                                    ;
Редактирование кода                 ; 
Ctrl + D	                        ; Выделить следующее совпадение (множественный курсор)
Alt + ⬆️ / ⬇️	                    ; Переместить текущую строку вверх/вниз
Shift + Alt + ⬆️ / ⬇️	            ; Дублировать строку вверх/вниз
Ctrl + L	                        ; Выделить всю строку
Shift + Delete                  	; Удалить строку
Ctrl + Enter	                    ; Вставить строку ниже
Ctrl + K Ctrl + K	                ; Установить/удалить метку
Ctrl + K Ctrl + N	                ; Перейти к следующей метке

                                    ;
Множественное форматирование        ; 
Ctrl + Alt + ⬆️ / ⬇️	            ; Добавить курсор выше/ниже

                                    ;
Рефакторинг                         ; 
Shift + Alt + F	                    ; Форматировать весь документ
Ctrl + K Ctrl + F	                ; Форматировать выделенный фрагмент
Ctrl + /	                        ; Закомментировать/раскомментировать строку
Shift + Alt + A	                    ; Блочный комментарий (/* */)
F2	                                ; Переименовать символ
Ctrl + .	                        ; Быстрые исправления (Quick Fix)
Ctrl + Shift + [	                ; Свернуть фрагмент кода
Ctrl + Shift + ]	                ; Развернуть фрагмент кода

                                    ;
Интерфейс                           ; 
Ctrl + B                        	; Показать/скрыть боковую панель
Ctrl + J	                        ; Показать/скрыть панель вывода/терминала
Ctrl + Shift + E	                ; Фокус на проводник (Explorer)
Ctrl + Shift + G                	; Фокус на Git
Ctrl + ,                        	; Открыть настройки
Ctrl + \                        	; Разделить редактор (новая группа)
Ctrl + Alt + ➡️                 	; Перейти к следующей группе редакторов
Ctrl + Q	                        ; Перейти к парной скобке
Ctrl + Shift + Q	                ; Выделить всё содержимое внутри скобок

                                    ;
Справка                             ; 
Ctrl + Shift + P	                ; Открыть палитру команд
Ctrl + M	                        ; Переключить режим "Zen" (макс. фокус)
