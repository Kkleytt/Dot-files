Перемещение; 
Ctrl + 0	            ; Переход в чат "Избранное"
Ctrl + [1-8]	        ; Переход в закрепленные чаты [1-8]
Ctrl + Tab	            ; Перейти к следующему чату в списке
Ctrl + Shift + Tab	    ; Перейти к предыдущему чату в списке
Alt + 0	                ; Папка "Все чаты"
Alt + [1-8]	            ; Переход по папкам
Ctrl + F	            ; Поиск по текущему чату

;
Форматирование;
Ctrl + K	            ; Добавить ссылку в сообщение
Ctrl + O	            ; Прикрепить файл к сообщению
Ctrl + B	            ; Жирный текст
Ctrl + I	            ; Курсивный текст
Ctrl + U	            ; Подчеркнутый текст
Ctrl + Shift + X	    ; Зачеркнутый текст
Ctrl + Shift + .	    ; Цитата
Ctrl + Shift + M	    ; Моноширинный текст
Ctrl + Shift + P	    ; Скрытый текст
Ctrl + Shift + N	    ; Убрать форматирование
